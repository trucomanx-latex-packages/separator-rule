\section{Install package}
Put the separator-rule.sty file in any of these locations
\begin{itemize}
\item Put the \texttt{separator-rule.sty} file in the same path of main tex file, or.
\item Execute the commmand:
\begin{mdframed}[backgroundcolor=black!10,rightline=false,leftline=false]
\begin{verbatim}
kpsewhich -var-value=TEXMFHOME
\end{verbatim}
\end{mdframed}
and this returns the path of local tex files. By example, if returns 
\begin{mdframed}[backgroundcolor=black!10,rightline=false,leftline=false]
\begin{verbatim}
/home/username/texmf
\end{verbatim}
\end{mdframed}
then, put the \texttt{separator-rule.sty} file in the directory.
\begin{mdframed}[backgroundcolor=black!10,rightline=false,leftline=false]
\begin{verbatim}
/home/username/texmf/tex/latex/separator-rule/separator-rule.sty
\end{verbatim}
\end{mdframed}
\end{itemize}

Next, I will provide examples that demonstrate the use of all the macros included in the package. 
This will help illustrate how each macro functions within the context of the package and provide 
practical insights into their applications.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Command: HRule }
\begin{verbatim}
\HRule[Height]
\end{verbatim}
\begin{mydescription}
 \item[Height:] Height of rule.
\end{mydescription}

\EXAMPLEOFSEPARATORRULE{\HRule{2pt}}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Command: HTextRule }
\begin{verbatim}
\HTextRule[Height]{Text}
\end{verbatim}
\begin{mydescription}
 \item[Height:] Height of rule.
 \item[Text:] Text in the middle of rule.
\end{mydescription}

\EXAMPLEOFSEPARATORRULE{\HTextRule[5pt]{My text}}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\subsection{Command: HObjectTextRule }

\begin{verbatim}
\HObjectTextRule[ObjectTextLength]{LeftObject}{Text}{RightObject}
\end{verbatim}
\begin{mydescription}
 \item[ObjectTextLength:] Length distance between any object and the text.
 \item[LeftObject:] Object that will be placed in the left side of text.
 \item[Text:] Text in the middle of rule.
 \item[RightObject:] Object that will be placed in the right side of text.
\end{mydescription}


\begin{mdframed}[backgroundcolor=black!10,rightline=false,leftline=false]
\begin{verbatim}
 \usepackage{pgfornament}
\end{verbatim}
\end{mdframed}

\EXAMPLEOFSEPARATORRULE{\HObjectTextRule{\pgfornament[width=2cm]{72}}{Text}{\pgfornament[width=2cm]{73}}}

\subsection{Command: HPgfoTextRule }
See more about pgfornament in your \href{https://ctan.org/pkg/pgfornament?lang=en}{homepage}.
\begin{verbatim}
\HPgfoTextRule[KeyVals]{LeftOrnament}{Text}
\end{verbatim}
\begin{mydescription}
 \item[KeyVals:] 
    \begin{description}
     \item[Separator=3pt] Length distance between any object and the text.
     \item[OrnamentWidth=2cm] Width of ornament.
     \item[OrnamentColor=black] Color of ornament.
     \item[TextColor=black] Color of text.
     \item[TextFormat=$\backslash$bfseries] Format of text.
     \item[BeforeSpace=-5pt] Space before the rule.
     \item[AfterSpace=5pt] Space after the rule.
    \end{description}
 \item[LeftOrnament:] Number of pgfornament that will be placed in the left side of text.
 \item[Text:] Text in the middle of rule.
\end{mydescription}

\EXAMPLEOFSEPARATORRULE{\HPgfoTextRule{87}{Text}}
